ords = [ 97, 110, 100, 114, 101, 97, 115, 32, 98, 97, 110, 104, 111, 108, 122, 101, 114 ]
chars = ords
  .reject { | n | n==32 }
  .map { | n | n.chr }
a, b = chars.slice!( 1...ords.index( 32 ) ), chars.slice( 2.. )
user_name = "a#{a.length}b#{b.length}"
